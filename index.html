// Zentraler App-Zustand (State) const appState = { selectedBikeId: null, selectedTrack: 'pannoniaring', bikes: [], // Liste aller gespeicherten Motorräder setups: {} // Gespeicherte Setups, verknüpft nach BikeId + Track }; // Beim Laden der Seite initialisieren document.addEventListener('DOMContentLoaded', () => { loadDataFromStorage(); initBikeUI(); }); // --- DATEN LADEN & SPEICHERN (LocalStorage) --- function loadDataFromStorage() { const savedBikes = localStorage.getItem('upperracing_bikes'); const savedSetups = localStorage.getItem('upperracing_setups'); if (savedBikes) appState.bikes = JSON.parse(savedBikes); if (savedSetups) appState.setups = JSON.parse(savedSetups); } function saveToStorage() { localStorage.setItem('upperracing_bikes', JSON.stringify(appState.bikes)); localStorage.setItem('upperracing_setups', JSON.stringify(appState.setups)); } // --- FAHRZEUG-LOGIK --- function initBikeUI() { // Hier kannst du die HTML-Elemente verknüpfen (Beispiel-IDs, passe sie einfach an dein HTML an) const bikeSelect = document.getElementById('bike-select'); const addBikeBtn = document.getElementById('add-bike-btn'); // Dropdown aktualisieren renderBikeDropdown(); // Event: Anderes Motorrad ausgewählt if (bikeSelect) { bikeSelect.addEventListener('change', (e) => { appState.selectedBikeId = e.target.value; loadSetupForCurrentSelection(); }); } // Event: Neues Motorrad hinzufügen (z.B. per Modal oder Eingabefeldern) if (addBikeBtn) { addbikeBtn.addEventListener('click', () => { // Beispiel-Werte aus deinen Input-Feldern auslesen const brand = document.getElementById('bike-brand-input').value; const model = document.getElementById('bike-model-input').value; const year = document.getElementById('bike-year-input').value; if (!brand || !model) { alert('Bitte Marke und Modell angeben!'); return; } const newBike = { id: 'bike_' + Date.now(), brand: brand, model: model, year: year }; appState.bikes.push(newBike); appState.selectedBikeId = newBike.id; saveToStorage(); renderBikeDropdown(); loadSetupForCurrentSelection(); }); } } function renderBikeDropdown() { const bikeSelect = document.getElementById('bike-select'); if (!bikeSelect) return; bikeSelect.innerHTML = ''; appState.bikes.forEach(bike => { const option = document.createElement('option'); option.value = bike.id; option.textContent = `${bike.brand} ${bike.model} (${bike.year || 'k.A.'})`; if (bike.id === appState.selectedBikeId) { option.selected = true; } bikeSelect.appendChild(option); }); } // --- SETUP-LOGIK --- function loadSetupForCurrentSelection() { if (!appState.selectedBikeId) return; // Eindeutiger Schlüssel für die Kombination aus Bike und Strecke const setupKey = `${appState.selectedBikeId}_${appState.selectedTrack}`; const currentSetup = appState.setups[setupKey] || {}; // Felder im HTML mit den gespeicherten Werten füllen (Beispiele) setInputValue('reifendruck-v', currentSetup.reifendruckV); setInputValue('reifendruck-h', currentSetup.reifendruckH); setInputValue('notizen-turn', currentSetup.notizen); } function saveCurrentSetup() { if (!appState.selectedBikeId) { alert('Bitte zuerst ein Motorrad auswählen!'); return; } const setupKey = `${appState.selectedBikeId}_${appState.selectedTrack}`; // Werte aus den HTML-Feldern einsammeln appState.setups[setupKey] = { reifendruckV: getInputValue('reifendruck-v'), reifendruckH: getInputValue('reifendruck-h'), notizen: getInputValue('notizen-turn'), updatedAt: new Date().toISOString() }; saveToStorage(); // Visuelles Feedback (z.B. "Erfolgreich gespeichert!" einblenden) showSaveSuccessMessage(); } // Hilfsfunktionen für Inputs function setInputValue(id, value) { const el = document.getElementById(id); if (el) el.value = value || ''; } function getInputValue(id) { const el = document.getElementById(id); return el ? el.value : ''; } function showSaveSuccessMessage() { const msg = document.getElementById('success-message'); if (msg) { msg.style.display = 'block'; setTimeout(() => { msg.style.display = 'none'; }, 2000); } }
